export const FamilyMatch = () => {
  const [matches, setMatches] = React.useState({});
  const [selected, setSelected] = React.useState(null);
  const [showResults, setShowResults] = React.useState(false);
  
  const pairs = [
    ['der Vater', 'father'],
    ['die Mutter', 'mother'],
    ['der Bruder', 'brother'],
    ['die Schwester', 'sister']
  ];

  const handleClick = (item) => {
    if (!selected) {
      setSelected(item);
    } else {
      setMatches(prev => ({
        ...prev,
        [selected]: item,
        [item]: selected
      }));
      setSelected(null);
    }
  };

  const checkAnswers = () => setShowResults(true);
  const reset = () => {
    setMatches({});
    setSelected(null);
    setShowResults(false);
  };

  return (
    <div className="matching-exercise">
      <div className="match-columns">
        <div className="german-column">
          {pairs.map(([german]) => (
            <button
              key={german}
              onClick={() => handleClick(german)}
              className={`match-button ${
                selected === german ? 'selected' : 
                german in matches ? 
                  (showResults ? 
                    (pairs.find(([g]) => g === german)[1] === matches[german] ? 'correct' : 'incorrect') 
                    : 'matched') 
                  : ''
              }`}
            >
              {german}
            </button>
          ))}
        </div>
        <div className="english-column">
          {pairs.map(([_, english]) => (
            <button
              key={english}
              onClick={() => handleClick(english)}
              className={`match-button ${
                selected === english ? 'selected' : 
                english in matches ? 
                  (showResults ? 
                    (pairs.find(([_, e]) => e === english)[0] === matches[english] ? 'correct' : 'incorrect') 
                    : 'matched')

              }`}
            >
              {english}
            </button>
          ))}

## 10. Memory Game (Vocabulary)

export const MemoryGame = () => {
  const [flippedIndexes, setFlippedIndexes] = React.useState([]);
  const [matched, setMatched] = React.useState([]);
  const [disabled, setDisabled] = React.useState(false);
  
  const cards = [
    { id: 1, content: 'Haus' },
    { id: 2, content: 'house' },
    { id: 3, content: 'Katze' },
    { id: 4, content: 'cat' },
    { id: 5, content: 'Buch' },
    { id: 6, content: 'book' },
    { id: 7, content: 'Hund' },
    { id: 8, content: 'dog' }
  ];

  // Create pairs matching logic
  const cardPairs = {
    'Haus': 'house',
    'house': 'Haus',
    'Katze': 'cat',
    'cat': 'Katze',
    'Buch': 'book',
    'book': 'Buch',
    'Hund': 'dog',
    'dog': 'Hund'
  };

  const handleCardClick = (index) => {
    // Prevent clicking if already two cards are flipped or card is already matched
    if (disabled || flippedIndexes.length === 2 || matched.includes(index)) {
      return;
    }

    // Prevent clicking the same card twice
    if (flippedIndexes.includes(index)) {
      return;
    }

    const newFlipped = [...flippedIndexes, index];
    setFlippedIndexes(newFlipped);

    // Check for matches when two cards are flipped
    if (newFlipped.length === 2) {
      setDisabled(true);
      const [firstIndex, secondIndex] = newFlipped;
      const firstCard = cards[firstIndex];
      const secondCard = cards[secondIndex];

      if (cardPairs[firstCard.content] === secondCard.content) {
        // Match found
        setMatched([...matched, firstIndex, secondIndex]);
        setFlippedIndexes([]);
        setDisabled(false);
      } else {
        // No match, flip cards back
        setTimeout(() => {
          setFlippedIndexes([]);
          setDisabled(false);
        }, 1000);
      }
    }
  };

  return (
    <div className="memory-game">
      <div className="memory-grid">
        {cards.map((card, index) => (
          <button
            key={index}
            onClick={() => handleCardClick(index)}
            disabled={disabled || matched.includes(index)}
            className={`memory-card ${
              flippedIndexes.includes(index) || matched.includes(index) ? 'flipped' : ''
            }`}
          >
            <div className="card-inner">
              <div className="card-front">?</div>
              <div className="card-back">{card.content}</div>
            </div>
          </button>
        ))}
      </div>
      <div className="exercise-buttons">
        <button
          onClick={() => {
            setFlippedIndexes([]);
            setMatched([]);
            setDisabled(false);
          }}
          className="reset-button"
        >
          New Game
        </button>
      </div>
    </div>
  );
};
